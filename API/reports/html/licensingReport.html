            function checkFirmwareVersion(fullVersionString) {
                version = fullVersionString;
                var start = version.indexOf("_V");
                var firstDot = version.indexOf(".", start);
                var secondDot = version.indexOf(".", firstDot + 1);
                        
                if (start == -1 || firstDot == -1 || secondDot == -1) {
                    return false;
                }
                        
                version = version.substring(start + "_V".length, secondDot);
                var splitVersion = version.split(".");
                if (splitVersion[0] < 3 ||
                    (splitVersion[0] == 3 &&
                     splitVersion[1] < 5)) {
                    return false;
                }
                
                return true;
